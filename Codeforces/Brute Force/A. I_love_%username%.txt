import java.util.*;
 
public class cosa {
 
	public static void main(String[] args) {
 
		Scanner en = new Scanner(System.in);
		int n, m, max, cont, min;
 
		n = en.nextInt();
		max = -1;
		cont = 0;
		m = en.nextInt();
		max = m;
		min = m;
		for (int i = 0; i < n - 1; i++) {
 
			m = en.nextInt();
			if (m > max) {
				max = m;
				cont++;
			} else if (m < min) {
				min = m;
				cont++;
			}
 
		}
		System.out.println(cont);
	}
 
}
