import java.util.*;
 
public class Testing {
	public static void main(String[] args) {
 
	
		Scanner sc=new Scanner(System.in);
		int n;
		String s;
		n=sc.nextInt();
		s=sc.next().toUpperCase();
		
		TreeSet<Character> ts = new TreeSet<>();
		for (int i = 0; i < n; i++) {
		ts.add(s.charAt(i));	
		}
		if(ts.size()==26) {
			System.out.println("YES");
		}
		else {
			System.out.println("NO");
		}
		
	}
	
}
